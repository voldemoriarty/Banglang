DIM DX AS NUMBER
DIM MEANING AS NUMBER
DIM CONDITION AS NUMBER
DIM FLAG AS NUMBER

SET DX AS 0.1
SET MEANING AS 4
SET CONDITION AS 0
SET FLAG AS 1

PRINT "This Program finds the meaning of life and Everything in the Universe"

WHILE FLAG = 1
	SET CONDITION AS (MEANING^0.5)-42
	IF CONDITION < 0 THEN
		SET CONDITION AS CONDITION*(0-1)
	ENDIF
	IF CONDITION < 0.05 THEN
		SET FLAG AS 0
	ENDIF
	SET MEANING AS MEANING+DX
WEND

PRINT "Meaning of Life and Everything in the Universe is (Aproximately):"
PRINT MEANING^0.5
